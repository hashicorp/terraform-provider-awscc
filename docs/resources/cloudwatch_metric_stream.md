# awscc_cloudwatch_metric_stream (Resource)

Resource Type definition for Metric Stream

## Example Usage

### Stream CloudWatch Metrics to S3 via Firehose

Sets up a CloudWatch Metric Stream that captures specific EC2 metrics (CPUUtilization, NetworkIn, NetworkOut) and forwards them through Kinesis Firehose to an S3 bucket, with all necessary IAM roles and permissions configured.

  role_name = "metrics-firehose-role"

      {
        Action = "sts:AssumeRole"

# Kinesis Firehose
resource "awscc_kinesisfirehose_delivery_stream" "example" {
  delivery_stream_name = "metric-stream-example"

  s3_destination_configuration = {
    bucket_arn = awscc_s3_bucket.example.arn
    role_arn   = awscc_iam_role.firehose.arn
    buffering_hints = {

      size_in_m_bs        = 64

  tags = [{
    key   = "Modified By"
    value = "AWSCC"

  role_arn      = awscc_iam_role.metric_stream.arn

    namespace = "AWS/EC2"

      "CPUUtilization",
